039    /**
040     *  A no-op unsupported terminal.
041     *
042     *  @author  <a href="mailto:mwp1@cornell.edu">Marc Prud'hommeaux</a>
043     */
044    public class UnsupportedTerminal
045            extends Terminal
046    {
047            public void initializeTerminal ()
048            {
049                    // nothing we need to do (or can do) for windows.
050            }
051    
052    
053            public boolean getEcho ()
054            {
055                    return true;
056            }
057    
058    
059            /**
060             *      Always returng 80, since we can't access this info on Windows.
061             */
062            public int getTerminalWidth ()
063            {
064                    return 80;
065            }
066    
067    
068            /**
069             *      Always returng 24, since we can't access this info on Windows.
070             */
071            public int getTerminalHeight ()
072            {
073                    return 80;
074            }
075    
076    
077            public boolean isSupported ()
078            {
079                    return false;
080            }
081    }
